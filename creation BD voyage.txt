-- Table VOYAGES
CREATE TABLE VOYAGES (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    DateVoyage DATE NOT NULL,
    Destination TEXT,
    Camion TEXT,
    Chauffeur TEXT
);

-- Table VOYAGE_LIGNES
CREATE TABLE VOYAGE_LIGNES (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    ID_VOYAGE INTEGER,
    Client TEXT,
    NumDossier TEXT,
    Quantite INTEGER,
    NbCarton INTEGER,
    NbPalette INTEGER,
    Termine BOOLEAN,
    FOREIGN KEY (ID_VOYAGE) REFERENCES VOYAGES(ID)
);
